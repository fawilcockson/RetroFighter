1    import tkinter
2    from tkinter import *
3    import os
4    import json
5    
6    
7    window = tkinter.Tk()
8    window.title("Retro Fighter")
9    window.geometry("1000x600")
10   window.resizable(True, True)
11   
12   # Create a dropdown list for the options for resolution
13   resolution = tkinter.StringVar()
14   resolution.set("Resolution")
15   resolutionMenu = tkinter.OptionMenu(window, resolution, "800x600", "1024x768", "1280x720", "1366x768", "1600x900",
16                                       "1920x1080")
17   print(resolution.get())
18   resolutionMenu.place(x=0, y=0)
19   
20   # Create a slider for the volume of the game
21   volume = tkinter.Scale(window, from_=0, to=100, orient=HORIZONTAL)
22   volume.place(x=0, y=50)
23   
24   # Create a checkbox for the fullscreen option
25   fullscreen = tkinter.IntVar()
26   fullscreen.set(0)
27   fullscreenCheck = tkinter.Checkbutton(window, text="Fullscreen", variable=fullscreen)
28   fullscreenCheck.place(x=0, y=100)
29   
30   # Create a set of print statements to print the values of the options
31   print("Resolution: " + resolution.get())
32   print("Volume: " + str(volume.get()))
33   print("Fullscreen: " + str(fullscreen.get()))
34   
35   # Create a button to return the main menu
36   backButton = tkinter.Button(window, text="Back", command=lambda: [window.destroy(), os.system('py menu.py')], height=5,width=5)
37   backButton.pack(anchor=tkinter.E)
38   
39   # Write the data to a JSON in a function
40   def writeJSON():
41       dataToJSON = {
42           "Resolution": resolution.get(),
43           "Volume": volume.get(),
44           "Fullscreen": fullscreen.get()
45       }
46       with open("retroFighter.cfg", "w") as outfile: outfile.write(json.dumps(dataToJSON))
47   
48   # Create a button to apply the settings by writing them to retroFighter.cfg
49   applyButton = tkinter.Button(window, text="Apply",
50                                command=lambda: [print("Applied"), writeJSON()], height=5,
51                                width=5)
52   applyButton.pack(anchor=tkinter.E)
53   
54   window.mainloop()
